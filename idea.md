function generateBasePath(nodeCount){
  let nodes = {}
  let id = 1

  nodes[id] = { id, next: [] }

  for(let i=0;i<nodeCount;i++){
    let newId = ++id
    nodes[newId] = { id:newId, next:[] }
    nodes[newId-1].next.push(newId)
  }

  return nodes
}
STEP2：分岐生成
function addBranches(nodes, branchProbability=0.3){
  Object.values(nodes).forEach(node=>{
    if(Math.random() < branchProbability){
      const newId = Object.keys(nodes).length + 1
      nodes[newId] = { id:newId, next:[] }
      node.next.push(newId)
    }
  })
}
STEP3：ループを作る（桃鉄感）
function addLoop(nodes){
  const ids = Object.keys(nodes)
  const a = ids[Math.floor(Math.random()*ids.length)]
  const b = ids[Math.floor(Math.random()*ids.length)]

  if(a !== b){
    nodes[a].next.push(Number(b))
  }
}
